Архитектура проекта
Компоненты системы:
Мозг (GPT):
Обрабатывает данные и генерирует стратегии.
Используются модели GPT-3.5 и GPT-4 в зависимости от сложности задач.
Мост:
Передаёт команды и данные между "Мозгом" и Binance API.
Выполняет обработку ошибок и логирование.
Binance API:
Предоставляет рыночные данные (цены, объёмы).
Выполняет команды на открытие и закрытие сделок.
Взаимодействие между компонентами:
Мозг → Мост: Генерация команд и запросов.
Мост → Binance API: Выполнение запросов и передача данных.
Binance API → Мост: Возвращение данных о рынке.
Мост → Мозг: Отправка результатов и обновлений.

using System.Net.Http;
using System.Text;
using System.Text.Json;
using System.Threading.Tasks;

namespace TradingConsoleApp.Modules
{
    public class GPTModule
    {
        private readonly string _apiKey;

        public GPTModule(string apiKey)
        {
            _apiKey = apiKey;
        }

        public async Task SendMessageAsync(string prompt)
        {
            using var client = new HttpClient();

            // Подготовка тела запроса
            var requestBody = new
            {
                model = "gpt-3.5-turbo",
                messages = new[]
                {
                    new { role = "user", content = prompt }
                }
            };

            // Сериализация JSON
            var json = JsonSerializer.Serialize(requestBody);
            var content = new StringContent(json, Encoding.UTF8, "application/json");

            // Установка заголовков
            client.DefaultRequestHeaders.Authorization =
                new System.Net.Http.Headers.AuthenticationHeaderValue("Bearer", _apiKey);

            // Выполнение запроса
            var response = await client.PostAsync("https://api.openai.com/v1/chat/completions", content);
            var responseContent = await response.Content.ReadAsStringAsync();

            if (response.IsSuccessStatusCode)
            {
                return responseContent;
            }
            else
            {
                throw new HttpRequestException($"GPT API Error: {response.StatusCode}, {responseContent}");
            }
        }
    }
}
